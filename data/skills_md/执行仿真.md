```python
import cloudpss
import time

# 获取模型的最新版本
revision = cloudpss_substation_Model.revision

# 选择第一个参数方案和第一个计算方案
job = cloudpss_substation_Model.jobs[0]
config = cloudpss_substation_Model.configs[0]

# 运行仿真任务
runner = revision.run(job, config)

# 等待仿真完成
while not runner.status():
    time.sleep(1)

# 获取仿真结果的plots曲线
plots = runner.result.getPlots()

# 输出plots曲线的数量
print(f"仿真结果包含 {len(plots)} 个曲线")
```
